187 void GameWindow::processCommand() {
188  QString errorMessage = "Invalid command. Type '?' for a list of possible commands.";
189  QString command = m_ui->textEdit->text().trimmed().toLower();
190 
191  if(command == '?') {
192  showHelp();
193  } else {
194  QStringList commandParts = command.split(' ');
195 
196  if(commandParts.size() == 4 && commandParts[0] == "go" && commandParts[1] == "to") {
197  bool okX, okY;
198  int x = commandParts[2].toInt(&okX);
199  int y = commandParts[3].toInt(&okY);
200 
201  if(okX && okY) {
202  m_controller->pathFinder(x, y);
203  }
204  } else if(commandParts.size() == 2) {
205  QString commandType = commandParts[0];
206  QString commandAction = commandParts[1];
207 
208  if(commandType == "move" && moveCommands.contains(commandAction)) {
209  moveCommands[commandAction].first();
210  } else if(commandType == "view" && viewCommands.contains(commandAction)) {
211  viewCommands[commandAction].first();
212  } else if(commandType == "game" && gameCommands.contains(commandAction)) {
213  gameCommands[commandAction].first();
214  } else if(commandType == "zoom" && zoomCommands.contains(commandAction)) {
215  zoomCommands[commandAction].first();
216  } else {
217  m_ui->plainTextEdit->setPlainText(errorMessage);
218  }
219  } else {
220  m_ui->plainTextEdit->setPlainText(errorMessage);
221  }
222  }
223  m_ui->textEdit->clear();
224 }
