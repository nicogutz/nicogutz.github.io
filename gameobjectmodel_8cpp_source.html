12 const QPointer<GameObject> GameObjectModel::getNeighbor(QPoint location, double direction, int offset) const {
13  // The other method of using simple trig was causing some issues with big offsets.
14  // This might be somewhat slower but works quite well. The whole game is based on directions relative to
15  // the GameObjects since this makes behaviors very easy to program and fits well with the way objects are displayed.
16  qreal qoffset = static_cast<qreal>(offset + 1);
17 
18  QTransform tf;
19  // We make a line with length 2*qoffset^2, since calculating the square root is slow and pointless.
20  // It just has to be longer than the biggest diagonal in a square.
21  auto line = tf.rotate(-direction).map(QLineF({0, 0}, {2 * qoffset * qoffset, 0}));
22 
23  QPointF pointF;
24  QLineF bounds {qoffset, -qoffset, qoffset, qoffset};
25  // We loop and check for an intersection with a line. The line is rotated until an intersection is found.
26  while(line.intersects(bounds, &pointF) != QLineF::BoundedIntersection) {
27  tf.reset();
28  bounds = tf.rotate(90).map(bounds);
29  }
30 
31  // This intersection is the grid location relative to the object.
32  auto point = pointF.toPoint();
33  int x = location.x() + point.x();
34  int y = location.y() + point.y();
35 
36  // No tile access allowed in the void.
37  if(0 > x || 0 > y || x >= getColumnCount() || y >= getRowCount()) {
38  return QPointer<GameObject>(nullptr);
39  }
40  return m_world.at(x).at(y);
41 }
