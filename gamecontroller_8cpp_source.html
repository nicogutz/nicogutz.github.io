138 void GameController::executePath(std::vector<int> path, bool full) {
139  // Tile at the start position
140  auto first_tile = qobject_cast<GameObject *>(m_protagonist->parent());
141  for(int move : path) {
142  // Assign the path tiles to DataRole Path
143  first_tile = first_tile->getNeighbor(((45 * move + 90) % 360));
144  first_tile->setData(DataRole::Path, true);
145  }
146 
147  for(int move : path) {
148  Direction direction = (Direction)((45 * move + 90) % 360);
149 
150  // Quick Delay for visualization
151  QTime dieTime = QTime::currentTime().addMSecs(100);
152  while(QTime::currentTime() < dieTime)
153  QCoreApplication::processEvents(QEventLoop::AllEvents, 100);
154 
155  if(direction != m_protagonist->getData(DataRole::Direction).value<Direction>()) {
156  characterMove(direction);
157  }
158 
159  // Check whether enemy is on the way of the path and attack it
160  if(auto tile = m_protagonist->getNeighbor(direction)) {
161  if(tile->hasChild({ObjectType::_ENEMIES_START, ObjectType::_ENEMIES_END})) {
162  automaticAttack();
163  }
164  }
165  // Play fully automatic
166  if(full) {
167  QPointer<const GameObject> obj;
168  // Find enemy or healthpack if energy or health too low. Number is sort of arbitrary
169  if(m_protagonist->getData(DataRole::Energy).toInt() < 80 || m_protagonist->getData(DataRole::PoisonLevel).toInt() > 15) {
170  obj = m_protagonist->nearest({ObjectType::_ENEMIES_START, ObjectType::_ENEMIES_END});
171 
172  } else if(m_protagonist->getData(DataRole::Health).toInt() < 80) {
173  obj = m_protagonist->nearest(ObjectType::HealthPack);
174  }
175  // Can be that there are no HP or enemies left.
176  if(obj) {
177  QPoint objPos = obj->getData(DataRole::Position).toPoint();
178  QPoint charPos = m_protagonist->getData(DataRole::Position).toPoint();
179  QPoint doorPos(m_models[m_gameLevel].first->getColumnCount(),
180  m_models[m_gameLevel].first->getRowCount());
181  int distObj = (objPos - charPos).manhattanLength();
182  int distDoor = (doorPos - charPos).manhattanLength();
183  // Check if the distance to the door is smaller than the distance to the object.
184  if(distObj < distDoor) {
185  pathFinder(objPos.x(), objPos.y());
186  }
187  // After we go to the object, return to the pathfinder function that called this.
188  // That function will schedule itself after this function finishes.
189  return;
190  }
191  }
192  characterMove(direction);
193  }

196 void GameController::pathFinder(int x, int y) {
197  bool full = (x == -1 && y == -1);
198  auto nodes = m_models[m_gameLevel].second; // Node class for the pathfinder
199 
200  int rows = m_models[m_gameLevel].first->getRowCount();
201  int cols = m_models[m_gameLevel].first->getColumnCount();
202 
203  // Get protagonist position in the world = start position of the pathfinder
204  auto pos = static_cast<GameObject *>(m_protagonist->parent())->getData(DataRole::Position).toPoint();
205 
206  Comparator<Node> comp = [](const Node &a, const Node &b) {
207  return a.h > b.h;
208  };
209  // Check for non valid input position
210  if(x >= cols || y >= rows || x < 0 || y < 0) {
211  y = rows - 1;
212  x = cols - 1;
213  }
214 
215  auto *start = &nodes[cols * pos.y() + pos.x()];
216  auto *dest = &nodes[cols * y + x];
217  PathFinder<Node, Node> pathFinder(nodes, start, dest, comp, cols, 0.001f);
218 
219  // Call the algorithm
220  auto path = pathFinder.A_star();
221 
222  executePath(path, full);
223 
224  // Run the method again in the next event loop if the game is on full auto.
225  if(full) {
